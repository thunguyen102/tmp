import fs from "node:fs";
import path from "node:path";

import { test } from "@/fixtures/role.fixture";
import MiraizAuthWorkflow from "@/workflows/miraiz/auth.workflow";
import { CREDENTIALS } from "@/constants";

// THROWAWAY exploration spec — read-only, reuses the existing proven login
// flow to get authenticated, then only observes (screenshot + ariaSnapshot).
// No form submit / state change beyond the login itself. Deleted after use.
test("explore post-login page (read-only)", async ({ miraizPage, mailPage }) => {
  await MiraizAuthWorkflow.ensureAuthenticatedSession(
    miraizPage,
    mailPage,
    {
      email: CREDENTIALS.MIRAIZ.EMAIL,
      password: CREDENTIALS.MIRAIZ.PASSWORD,
    },
    {
      email: CREDENTIALS.MAIL_TM.EMAIL,
      password: CREDENTIALS.MAIL_TM.PASSWORD,
    },
  );

  const outDir = path.resolve(process.cwd(), "explore-output-tmp");
  fs.mkdirSync(outDir, { recursive: true });

  console.log("URL after login:", miraizPage.url());
  console.log("TITLE after login:", await miraizPage.title());

  await miraizPage.screenshot({
    path: path.join(outDir, "post-login-home.png"),
    fullPage: true,
  });

  const snapshot = await miraizPage.locator("body").ariaSnapshot();
  fs.writeFileSync(path.join(outDir, "post-login-home-snapshot.yaml"), snapshot);
});
